document.getElementById('uploadEmoji').addEventListener('click', () => { const fileInput = document.getElementById('emojiFile'); const emojiName = document.getElementById('emojiName').value; if (!fileInput.files[0] || !emojiName) return alert('Please select a file and provide a name.'); const reader = new FileReader(); reader.onload = () => { const base64 = reader.result.split(',')[1]; const emojiData = { name: emojiName, image: `data:image/png;base64,${base64}` }; document.getElementById('emojiOutput').textContent = JSON.stringify(emojiData, null, 2); }; reader.readAsDataURL(fileInput.files[0]); });

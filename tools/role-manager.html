document.getElementById('generateRole').addEventListener('click', () => { const roleName = document.getElementById('roleName').value; const roleColor = document.getElementById('roleColor').value; if (!roleName || !roleColor) return alert('Please provide both name and color.'); const roleData = { name: roleName, color: roleColor }; document.getElementById('roleOutput').textContent = JSON.stringify(roleData, null, 2); });

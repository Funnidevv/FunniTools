document.getElementById('sendWebhook').addEventListener('click', async () => { const url = document.getElementById('webhookUrl').value; const content = document.getElementById('webhookMessage').value; if (!url || !content) return alert('Please provide both URL and message.'); try { await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ content }) }); alert('Message sent!'); } catch (error) { alert('Failed to send message.'); } });

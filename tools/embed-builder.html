document.getElementById('sendEmbed').addEventListener('click', async () => { const url = document.getElementById('embedWebhookUrl').value; const title = document.getElementById('embedTitle').value; const description = document.getElementById('embedDescription').value; if (!url || !title || !description) return alert('Please fill in all fields.'); const embed = { title, description, color: 0x5865F2 }; try { await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ embeds: [embed] }) }); alert('Embed sent!'); } catch (error) { alert('Failed to send embed.'); } });
